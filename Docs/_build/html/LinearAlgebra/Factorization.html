def Solve(A,b):
    """
    Ax=b
    L(L^T)x=b
    Ly=b
    (L^T)x=y
    """
    #Init vector to store solutions of the linear systems
    y = vec(b.len())
    x = vec(b.len())
    LT = mat(b.len(),b.len())
    #We perform Cholesky factorization
    L = Cholesky(A)
    LT = L.T()
    #Init the linear system Ly=b
    S1 = LinSys(L,b)
    S1.setType("LOWER")
    #Solve linear system using forward sub
    y=S1.ForwardSub()
    #Init the linear system (L^T)x=y
    S2 = LinSys(LT,y)
    S2.setType("UPPER")
    #Solve linear system using backward sub
    x = S2.BackSub()
    #Freeing memory
    LT.free()
    L.free()
    y.free()
    return x;
